dados$faixa_etaria <- ifelse(dados$idade > 18 & dados$idade < 30, "18-29", 
       ifelse(dados$idade > 29 & dados$idade < 40, "30-39",
       ifelse(dados$idade > 39 & dados$idade < 50, "40-49",
       ifelse(dados$idade > 49 & dados$idade < 60, "50-59",
       ifelse(dados$idade > 59 & dados$idade < 70, "60-69",
       ifelse(dados$idade > 69, "70+", NA))))))

dados$grupo_risco <- ifelse(

  "Alto risco", 
  "Baixo risco")
